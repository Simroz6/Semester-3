#include <iostream>
using namespace std;

int binSearch(int arr[], int n, int target) {
   int left = 0, right = n - 1;
   while (left <= right) {
      int mid = left + (right - left) / 2; //Here you can also use floor value of low+hi/2
      if (arr[mid] == target)
         return mid;
      else if (arr[mid] < target)
         left = mid + 1;
      else
         right = mid - 1;
   }
   return -1; 
}

int main() {
   int arr[] = {1, 3, 5, 7, 9};
   int n = sizeof(arr) / sizeof(arr[0]);
   int target = 7;
   cout << "Given array: ";
   for (int i = 0; i < n; i++) {
      cout << arr[i] << " ";
   }
   cout << endl;
   cout << "Target Element to search: " << target 
        << endl;
   int result = binSearch(arr, n, target);
   cout << "Element " << target << " found at index: " 
        << result << endl;
   return 0;
}
